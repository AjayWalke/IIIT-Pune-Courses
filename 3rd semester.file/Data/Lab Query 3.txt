 Code NUMBER primary key,
 Location VARCHAR(255),
 Capacity NUMBER(10)
);
CREATE TABLE Boxes (
 Code VARCHAR(255) primary key,
 Contents VARCHAR(255),
 Value NUMBER(10),
 Warehouse NUMBER(10) REFERENCES Warehouses(Code)
);

DESC Boxes;
